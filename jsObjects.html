//js properties let car = { make: "Toyota", model: "Corolla", color: "Red", speed: 120 }; console.log(car); //output the whole object console.log(car.make); //accessing a property //js values let laptop = { brand: "Dell", price: 800, isNew: true }; console.log(laptop.brand); //Dell console.log(laptop.price) //800 console.log(laptop.isNew); //true //js Dynamic Properties let key = "color"; let bike = { brand: "Yamaha", (key): "Black" }; console.log(bike.color); //Black
